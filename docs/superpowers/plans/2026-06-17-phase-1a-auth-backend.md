export const envSchema = z.object({
  NODE_ENV: z.enum(["development", "test", "production"]).default("development"),
  PORT: z.coerce.number().default(4000),
  DATABASE_URL: z.string().url(),
  REDIS_URL: z.string().url(),

  JWT_ACCESS_TTL: z.string().default("15m"),
  JWT_REFRESH_TTL: z.string().default("7d"),
  CORS_ORIGIN: z.string().default("http://localhost:3000"),
});

export type Env = z.infer<typeof envSchema>;

/** Validate process.env at startup; throws if invalid. */
export function validateEnv(): Env {
  const parsed = envSchema.safeParse(process.env);
  if (!parsed.success) {
    const errors = parsed.error.issues.map((i) => `${i.path.join(".")}: ${i.message}`).join("; ");
    throw new Error(`Invalid environment configuration: ${errors}`);
  }
  return parsed.data;
}

```typescript
import { Inject, Injectable } from "@nestjs/common";
import { ConfigService } from "@nestjs/config";
import { JwtService } from "@nestjs/jwt";
import { randomUUID } from "node:crypto";
import Redis from "ioredis";
import { REDIS_CLIENT } from "../../redis/redis.constants";

interface TokenPayload {
  sub: string;
  role: string;
  companyId: string;
  jti: string;
}

@Injectable()
export class TokenService {
  private readonly refreshTtlSeconds: number;

  constructor(
    private readonly jwt: JwtService,
    private readonly config: ConfigService,
    @Inject(REDIS_CLIENT) private readonly redis: Redis,
  ) {
    // Parse "7d" / "15m" / "3600s" into seconds for Redis TTL.
    this.refreshTtlSeconds = this.parseTtl(config.get<string>("JWT_REFRESH_TTL") ?? "7d");
  }

  issueAccess(user: { id: string; role: string; companyId: string }): string {
    const payload: TokenPayload = {
      sub: user.id,
      role: user.role,
      companyId: user.companyId,
      jti: randomUUID(),
    };
    return this.jwt.sign(payload, {
      secret: this.config.get<string>("JWT_SECRET"),
      expiresIn: this.config.get<string>("JWT_ACCESS_TTL") ?? "15m",
    });
  }

  issueRefresh(user: { id: string; role: string; companyId: string }): string {
    const payload: TokenPayload = {
      sub: user.id,
      role: user.role,
      companyId: user.companyId,
      jti: randomUUID(),
    };
    return this.jwt.sign(payload, {
      secret: this.config.get<string>("JWT_SECRET"),
      expiresIn: this.config.get<string>("JWT_REFRESH_TTL") ?? "7d",
    });
  }

  /** Verify signature + expiry; returns payload or throws. */
  verify(token: string): TokenPayload {
    return this.jwt.verify<TokenPayload>(token, { secret: this.config.get<string>("JWT_SECRET") });
  }

  /** Deny a refresh jti so it can no longer be used for rotation. TTL = token's remaining life. */
  async revokeRefresh(jti: string): Promise<void> {
    await this.redis.set(this.denyKey(jti), "1", "EX", this.refreshTtlSeconds);
  }

  async isRefreshRevoked(jti: string): Promise<boolean> {
    const result = await this.redis.get(this.denyKey(jti));
    return result !== null;
  }

  private denyKey(jti: string): string {
    return `rev:refresh:${jti}`;
  }

  private parseTtl(ttl: string): number {
    const match = /^(\d+)([smhd])$/.exec(ttl);
    if (!match) return 7 * 24 * 60 * 60;
    const value = Number(match[1]);
    const unit = match[2];
    const multipliers: Record<string, number> = { s: 1, m: 60, h: 3600, d: 86400 };
    return value * (multipliers[unit] ?? 86400);
  }
}
```

- [ ] **Step 1: Write `apps/api/src/auth/auth.service.ts`**

```typescript
import { ConflictException, Inject, Injectable, UnauthorizedException } from "@nestjs/common";
import bcrypt from "bcrypt";
import { PrismaService } from "../prisma/prisma.service";
import { TokenService } from "./token/token.service";
import type { AuthUser, LoginInput, RegisterInput } from "@iam/shared";

const BCRYPT_COST = 12;

@Injectable()
export class AuthService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly tokens: TokenService,
  ) {}

  async register(input: RegisterInput): Promise<{ user: AuthUser; accessToken: string; refreshToken: string }> {
    const existing = await this.prisma.user.findUnique({ where: { email: input.email } });
    if (existing) throw new ConflictException("Email already registered");

    const passwordHash = await bcrypt.hash(input.password, BCRYPT_COST);

    const user = await this.prisma.user.create({
      data: {
        email: input.email,
        password: passwordHash,
        firstName: input.firstName,
        lastName: input.lastName,
        role: "admin",
        company: { create: { name: input.company } },
      },
    });

    return this.toAuthResponse(user);
  }

  async login(input: LoginInput): Promise<{ user: AuthUser; accessToken: string; refreshToken: string }> {
    const user = await this.prisma.user.findUnique({ where: { email: input.email } });
    if (!user) throw new UnauthorizedException("Invalid credentials");

    const ok = await bcrypt.compare(input.password, user.password);
    if (!ok) throw new UnauthorizedException("Invalid credentials");

    return this.toAuthResponse(user);
  }

  /** Rotate: verify refresh, revoke its jti, issue a new pair. */
  async refresh(oldRefresh: string): Promise<{ accessToken: string; refreshToken: string }> {
    let payload: { sub: string; role: string; companyId: string; jti: string };
    try {
      payload = this.tokens.verify(oldRefresh);
    } catch {
      throw new UnauthorizedException("Invalid refresh token");
    }
    if (await this.tokens.isRefreshRevoked(payload.jti)) {
      throw new UnauthorizedException("Refresh token revoked");
    }
    await this.tokens.revokeRefresh(payload.jti);

    const accessToken = this.tokens.issueAccess({ id: payload.sub, role: payload.role, companyId: payload.companyId });
    const refreshToken = this.tokens.issueRefresh({ id: payload.sub, role: payload.role, companyId: payload.companyId });
    return { accessToken, refreshToken };
  }

  async logout(refreshToken: string): Promise<void> {
    try {
      const payload = this.tokens.verify(refreshToken);
      await this.tokens.revokeRefresh(payload.jti);
    } catch {
      // Idempotent: an already-invalid token is effectively logged out.
    }
  }

  async me(userId: string): Promise<AuthUser> {
    const user = await this.prisma.user.findUniqueOrThrow({ where: { id: userId } });
    return this.toAuthUser(user);
  }

  private async toAuthResponse(user: {
    id: string; email: string; firstName: string; lastName: string; role: string; companyId: string;
  }): Promise<{ user: AuthUser; accessToken: string; refreshToken: string }> {
    const authUser: AuthUser = {
      id: user.id, email: user.email, firstName: user.firstName,
      lastName: user.lastName, role: user.role as AuthUser["role"], companyId: user.companyId,
    };
    const accessToken = this.tokens.issueAccess(authUser);
    const refreshToken = this.tokens.issueRefresh(authUser);
    return { user: authUser, accessToken, refreshToken };
  }

  private toAuthUser(user: {
    id: string; email: string; firstName: string; lastName: string; role: string; companyId: string;
  }): AuthUser {
    return {
      id: user.id, email: user.email, firstName: user.firstName,
      lastName: user.lastName, role: user.role as AuthUser["role"], companyId: user.companyId,
    };
  }
}
```

import { CanActivate, ExecutionContext, Injectable } from "@nestjs/common";
import { Reflector } from "@nestjs/core";
import { ROLES_KEY } from "../decorators/roles.decorator";

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}
  canActivate(context: ExecutionContext): boolean {
    const required = this.reflector.getAllAndOverride<string[] | undefined>(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (!required || required.length === 0) return true;
    const { user } = context.switchToHttp().getRequest();
    return !!user && required.includes(user.role);
  }
}

/** Extracts refresh token from either the httpOnly cookie or the Authorization body field. */
const cookieOrBodyExtractor = (req: any): string | null => {
  if (req?.cookies?.refresh_token) return req.cookies.refresh_token;
  if (req?.body?.refreshToken) return req.body.refreshToken;
  return null;
};

@Injectable()
export class JwtRefreshStrategy extends PassportStrategy(
  {
    jwtFromRequest: cookieOrBodyExtractor,
    passReqToCallback: true,
    secretOrKey: new ConfigService().get<string>("JWT_SECRET")!,
  } as any,
  "jwt-refresh",
) {
  constructor(config: ConfigService) {
    super({
      jwtFromRequest: cookieOrBodyExtractor,
      passReqToCallback: true,
      secretOrKey: config.get<string>("JWT_SECRET")!,
    } as any);
  }
  validate(req: any, payload: any) {
    if (!payload) throw new UnauthorizedException();
    return payload;
  }
}

    const result = await this.auth.register(input);
    res.cookie("refresh_token", result.refreshToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === "production",
      sameSite: "lax",
      path: "/auth",
    });
    return result;
  }

  @Post("login")
  @Throttle({ default: { limit: 10, ttl: 60_000 } })
  @HttpCode(200)
  async login(@Body() body: unknown, @Res({ passthrough: true }) res: Response) {
    const input = loginSchema.parse(body);
    const result = await this.auth.login(input);
    res.cookie("refresh_token", result.refreshToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === "production",
      sameSite: "lax",
      path: "/auth",
    });
    return result;
  }

  @Post("refresh")
  @HttpCode(200)
  async refresh(@Req() req: any) {
    const refreshToken = req.cookies?.refresh_token ?? req.body?.refreshToken;
    return this.auth.refresh(refreshToken);
  }

  @Post("logout")

- [ ] **Step 1: Write ADR 0002**

```markdown
# ADR 0002: Refresh-Token Rotation + httpOnly Cookie

- **Status:** Accepted
- **Date:** 2026-06-17
- **Phase:** 1a

## Context
JWT access tokens are short-lived (15m) and verified statelessly. Refresh tokens
(7d) must be revocable so logout and theft-detection work. Storing both tokens
in localStorage exposes them to XSS.

## Decision
- Access token: in-memory only on the client (lost on reload → silent refresh).
- Refresh token: JWT with a `jti`, stored in an httpOnly, sameSite=lax cookie.
- **Rotation:** every `/auth/refresh` issues a fresh access+refresh pair with a
  new `jti`; the consumed refresh's `jti` is added to a Redis denylist with TTL
  equal to its remaining lifetime. Reuse of a revoked refresh → 401 (theft signal).
- `/auth/logout` revokes the current refresh `jti`.

## Consequences
- Stolen access token: valid for ≤15m only.
- Stolen refresh token: invalidated on first legitimate refresh; attacker reuse
  triggers 401, signalling compromise.
- Production cross-origin (Vercel → Render) requires `sameSite: 'none'` +
  `secure: true` on the cookie (configured in Phase 10 deployment).
- Access token is NOT denied via Redis — that would negate statelessness. Its
  short TTL is the accepted exposure window.
```
